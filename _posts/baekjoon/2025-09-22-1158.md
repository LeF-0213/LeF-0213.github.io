## Python
```python
from collections import deque

def solution(N, K):
    q = deque(i for i in range(1, N + 1))
    answer = []
    
    while(q):
        for _ in range(K - 1):
            q.append(q.popleft())
        answer.append(q.popleft())
        
    print(f"<{', '.join(map(str, answer))}>")
          
N, K = map(int, input().split())
solution(N, K)
```

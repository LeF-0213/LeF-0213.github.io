## Java
```java
class Solution {
    public int[] solution(int brown, int yellow) {
        int[] answer = new int[2];
        int carpet = brown + yellow;
        int limit = (int)Math.sqrt(carpet);
        
        for(int h = 2; h <= limit; h++) {
            if(carpet % h == 0) {
                int w = carpet / h;
                if((w + h - 2) * 2 == brown) {
                    answer[0] = w;
                    answer[1] = h;
                    break;
                }
            }
        }
        
        return answer;
    }
}
```

## Java
```java
import java.util.HashMap;

class Solution {
    public int[] solution(String s) {
        int[] answer = new int[s.length()];
        HashMap<Character, Integer> last = new HashMap<>();
        
        for(int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            answer[i] = i - last.getOrDefault(c, i + 1);
            last.put(c, i);
        }
        
        return answer;
    }
}
```

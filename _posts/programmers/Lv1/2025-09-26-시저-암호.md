## Java
```java
class Solution {
    public String solution(String s, int n) {
        String answer = "";
        n = n % 26;
        
        for(char c : s.toCharArray()) {
            if(Character.isUpperCase(c)) {
                answer += (char)((c - 'A' + n) % 26 + 'A');
            } else if(Character.isLowerCase(c)) {
                answer += (char)((c - 'a' + n ) % 26 + 'a');
            } else {
                answer += c;
            }
        }
        
        return answer;
    }
}
```
